JavaScript ES5 Nesne Yöntemleri

ECMAScript 5, JavaScript'e birçok yeni Nesne Yöntemi ekledi.

ES5 Yeni Nesne Yöntemleri

// Bir nesne özelliği ekleme veya değiştirme

Object.defineProperty(object, property, descriptor)


// Birçok nesne özelliği ekleme veya değiştirme

Object.defineProperties(object, descriptors)


// Özelliklere Erişim

Object.getOwnPropertyDescriptor(object, property)


// Tüm özellikleri dizi olarak döndürür

Object.getOwnPropertyNames(object)


// Sayılabilir özellikleri bir dizi olarak döndürür

Object.keys(object)


// Prototipe erişim

Object.getPrototypeOf(object)


// Bir nesneye özellik eklenmesini önler

Object.preventExtensions(object)

// Nesneye özellikler eklenebilirse true değerini döndürür

Object.isExtensible(object)


// Nesne özelliklerinde değişiklik yapılmasını önler (değerlerde değil)

Object.seal(object)

// Nesne mühürlenirse true değerini döndürür

Object.isSealed(object)


// Nesnedeki değişiklikleri önler

Object.freeze(object)

// Nesne dondurulmuşsa true değerini döndürür

Object.isFrozen(object)


Özellik Değerini Değiştirme

Sözdizimi

Object.defineProperty (nesne, özellik, {value: value})

Bu örnek bir özellik değerini değiştirir:

var person = {

firstName: "John",

lastName : "Doe",

language : "EN"

};


// Bir özelliği değiştirme

Object.defineProperty(person, "language", {value : "NO"});

Meta Verileri Değiştirme

ES5, aşağıdaki özellik meta verilerinin değiştirilmesine izin verir:

writable: true // Özellik değeri değiştirilebilir

numaralandırılabilir: true // Özellik numaralandırılabilir

configurable: true // Özellik yeniden yapılandırılabilir

writable: false // Özellik değeri değiştirilemez

numaralandırılabilir: false // Özellik numaralandırılamaz

configurable: false // Özellik yeniden yapılandırılamaz

ES5 alıcıların ve ayarlayıcıların değiştirilmesine izin verir:

// Alıcıyı tanımlama

get: function() { return language }

// Bir ayarlayıcı tanımlama

set: function(value) { language = value }

Bu örnek, dili salt okunur yapar:

Object.defineProperty(person, "language",

{writable:false});

Bu örnek dili sayılmaz kılar:

Object.defineProperty(person, "language",

{enumerable:false});

Tüm Özellikleri Listeleme

Bu örnek, bir nesnenin tüm özelliklerini listeler:

var person = {

firstName: "John",

lastName : "Doe"

language : "EN"

};


Object.defineProperty(person, "language", {enumerable:false});

Object.getOwnPropertyNames(person); // Returns an array of properties

Numaralandırılabilir Özellikleri Listeleme

Bu örnek, yalnızca bir nesnenin numaralandırılabilir özelliklerini listeler:

var person = {

firstName: "John",

lastName : "Doe"

language : "EN"

};


Object.defineProperty(person, "language", {enumerable:false});

Object.keys(person); // Bir dizi numaralandırılabilir özellik döndürür

Mülk Ekleme

Bu örnek, bir nesneye yeni bir özellik ekler:

// Nesne oluşturma:

var person = {

firstName: "John",

lastName : "Doe",

language : "EN"

};


// Özellik ekle

Object.defineProperty(person, "year", {value:"2008"});

Getter ve Setter Ekleme

Object.defineProperty () yöntemi, Getters ve Setters eklemek için de kullanılabilir:

// Nesne oluşturma

var person = {firstName:"John", lastName:"Doe"};


// Alıcıyı tanımla

Object.defineProperty(obj, "reset", {

get : function () {this.counter = 0;}

});

Object.defineProperty(obj, "increment", {

get : function () {this.counter++;}

});

Object.defineProperty(obj, "decrement", {

get : function () {this.counter--;}

});

Object.defineProperty(obj, "add", {

set : function (value) {this.counter += value;}

});

Object.defineProperty(obj, "subtract", {

set : function (i) {this.counter -= i;}

});


// Sayaçla oynayın:

obj.reset;

obj.add = 5;

obj.subtract = 1;

obj.increment;

obj.decrement;
